// ============================================== JANELA COM DOIS BOTOES E FEEDBACK DE QUAL BOTÃO FOI PRESSIONADO.

import javax.swing.*;
import java.awt.FlowLayout;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

public class ExemploJanela {
    public static void main(String[] args) {
        // 1. Criar o frame (janela)
        JFrame janela = new JFrame("Minha Janela");
        janela.setSize(400, 300);
        janela.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        janela.setLayout(new FlowLayout()); // Organiza os botões em linha

        // 2. Criar botões
        JButton botao1 = new JButton("Botão 1");
        JButton botao2 = new JButton("Botão 2");

        // 3. Adicionar ação ao botão 1
        botao1.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                System.out.println("Botao 1 clicado!");
                
            }
        });
            
        // 4. Adicionar ação ao botão 2
        botao2.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                System.out.println("Botao 2 clicado!");
            }
        });

        // 5. Adicionar botões à janela
        janela.add(botao1);
        janela.add(botao2);

        // 6. Tornar a janela visível
        janela.setVisible(true);
    }
}
